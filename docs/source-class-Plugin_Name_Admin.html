341:      * Bubble Notification for pending cpt<br>
342:      * NOTE: add in $post_types your cpts<br>
343:      *
344:      *        Reference:  http://wordpress.stackexchange.com/questions/89028/put-update-like-notification-bubble-on-multiple-cpts-menus-for-pending-items/95058
345:      *
346:      * @since    1.0.0
347:      */
348:     function pending_cpt_bubble() {
349:         global $menu;
350: 
351:         $post_types = $this->cpts;
352:         foreach ( $post_types as $type ) {
353:             if ( !post_type_exists( $type ) ) {
354:                 continue;
355:             }
356:             // Count posts
357:             $cpt_count = wp_count_posts( $type );
358: 
359:             if ( $cpt_count->pending ) {
360:                 // Menu link suffix, Post is different from the rest
361:                 $suffix = ( 'post' == $type ) ? '' : "?post_type=$type";
362: 
363:                 // Locate the key of 
364:                 $key = self::recursive_array_search_php( "edit.php$suffix", $menu );
365: 
366:                 // Not found, just in case 
367:                 if ( !$key ) {
368:                     return;
369:                 }
370: 
371:                 // Modify menu item
372:                 $menu[ $key ][ 0 ] .= sprintf(
373:                         '<span class="update-plugins count-%1$s"><span class="plugin-count">%1$s</span></span>', $cpt_count->pending
374:                 );
375:             }
376:         }

386:     private function recursive_array_search_php( $needle, $haystack ) {
387:         foreach ( $haystack as $key => $value ) {
388:             $current_key = $key;

468:     /**
469:      * Process a settings import from a json file
470:      * @since    1.0.0
471:      */
472:     function settings_import() {
473: 
474:         if ( empty( $_POST[ 'pn_action' ] ) || 'import_settings' != $_POST[ 'pn_action' ] ) {
475:             return;
476:         }
477: 
478:         if ( !wp_verify_nonce( $_POST[ 'pn_import_nonce' ], 'pn_import_nonce' ) ) {
479:             return;
480:         }
481: 
482:         if ( !current_user_can( 'manage_options' ) ) {
483:             return;
484:         }
485:         $extension = end( explode( '.', $_FILES[ 'import_file' ][ 'name' ] ) );
486: 
487:         if ( $extension != 'json' ) {
488:             wp_die( __( 'Please upload a valid .json file', $this->plugin_slug ) );
489:         }
490: 
491:         $import_file = $_FILES[ 'import_file' ][ 'tmp_name' ];
492: 
493:         if ( empty( $import_file ) ) {
494:             wp_die( __( 'Please upload a file to import', $this->plugin_slug ) );
495:         }
496: 
497:         // Retrieve the settings from the file and convert the json object to an array.
498:         $settings = ( array ) json_decode( file_get_contents( $import_file ) );
499: 
500:         update_option( $this->plugin_slug . '-settings', get_object_vars( $settings[ 0 ] ) );
501:         update_option( $this->plugin_slug . '-settings-second', get_object_vars( $settings[ 1 ] ) );
502: 
503:         wp_safe_redirect( admin_url( 'options-general.php?page=' . $this->plugin_slug ) );
504:         exit;
505:     }
