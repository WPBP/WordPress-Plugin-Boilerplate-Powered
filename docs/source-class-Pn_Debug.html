 41:     /**
 42:      * Debugs a variable
 43:      * Only visible to admins if WP_DEBUG is on
 44:      * @param mixed $var the var to debug
 45:      * @param bool $die (optional) whether to die after outputting
 46:      * @param string $function (optional) the function to call, usually either print_r or var_dump, but can be anything
 47:      * @return unknown
 48:      */
 49:     function log( $var, $die = false, $function = 'var_dump' ) {

 51:         if ( !current_user_can( 'manage_options' ) || !WP_DEBUG ) {
 52:             return;
 53:         }
 54: 
 55:         ob_start();
 56:         echo "<!-- BEGIN DEBUG OUTPUT --><PRE>\n";
 57:         call_user_func( $function, $var );
 58:         echo "\n</PRE><!-- END DEBUG OUTPUT -->\n";

 60:         if ( $die ) {
 61:             die();
 62:         }
 63: 
 64:         $debug = ob_get_clean();
 65: 
 66:         $this->history[] = $debug;
 67: 
 68:         //allow this to be used as a filter
 69:         return $var;
 70:     }
